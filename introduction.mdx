# Welcome to Erdo

Erdo is a powerful platform for building AI agents that can automate complex workflows, integrate with external services, and make intelligent decisions. With our Python SDK, you can create sophisticated automation that combines language models, code execution, web research, and custom integrations.

## Quick Example

Here's a complete AI agent that analyzes data and makes intelligent decisions:

```python
from erdo import Agent
from erdo.actions import llm, memory, codeexec, utils
from erdo.conditions import IsSuccess, GreaterThan, IsError, TextContains

# Create your agent
data_analyst = Agent(
    name="smart_data_analyst",
    description="Analyzes data and provides intelligent insights with validation"
)

# Step 1: Analyze the data with structured output
analysis_step = data_analyst.step(
    llm.message(
        model="claude-sonnet-4",
        query="Analyze this dataset and provide insights: {{dataset}}",
        response_format={
            "Type": "json_schema",
            "Schema": {
                "type": "object",
                "properties": {
                    "insights": {"type": "array"},
                    "confidence": {"type": "number"},
                    "recommendations": {"type": "array"},
                    "status": {"type": "string"}
                }
            }
        }
    )
)

# Step 2: Parse JSON format
validation_step = data_analyst.step(
    utils.parse_json(
        json="{{analysis_step.output.response}}"
    ),
    depends_on=analysis_step
)

# Step 3: Generate code for deeper analysis
code_generation = data_analyst.step(
    llm.message(
        model="claude-sonnet-4",
        query="Generate Python code to validate these insights: {{validation_step.output.insights}}",
        system_prompt="You are an expert data scientist. Generate clean, efficient Python code."
    ),
    depends_on=validation_step
)

# Step 4: Execute the generated code
code_execution = data_analyst.step(
    codeexec.execute(
        code_files=[{
            "filename": "validation.py",
            "content": "{{code_generation.output.code}}"
        }]
    ),
    depends_on=code_generation
)

# Validate analysis contains expected content
analysis_step.on(
    IsSuccess() & TextContains("insights") & TextContains("confidence"),
    memory.store(memory={
        "content": "{{analysis_step.output.response}}",
        "type": "validated_analysis",
        "tags": ["analysis", "validated", "{{domain}}"]
    })
)

# Validate parsed JSON has required fields
validation_step.on(
    IsSuccess() & TextContains("insights") & TextContains("confidence") & TextContains("recommendations"),
    memory.store(memory={
        "content": "{{validation_step.output}}",
        "type": "validated_json",
        "tags": ["analysis", "validated", "{{domain}}"]
    })
)

# Handle high-confidence results with additional validation
validation_step.on(
    IsSuccess() & GreaterThan("confidence", "0.8") & TextContains("status", "complete"),
    memory.store(memory={
        "content": "{{validation_step.output.insights}}",
        "type": "verified_insights",
        "tags": ["analysis", "high-confidence", "{{domain}}"],
        "extra": {
            "confidence": "{{validation_step.output.confidence}}",
            "recommendations": "{{validation_step.output.recommendations}}"
        }
    })
)

# Validate code generation output format
code_generation.on(
    IsSuccess() & TextContains("def ") & TextContains("import "),
    utils.send_status(
        status="code_validated",
        message="Generated code contains proper function definitions and imports"
    )
)

# Handle malformed code generation
code_generation.on(
    IsSuccess() & ~TextContains("def "),
    llm.message(
        model="claude-sonnet-4",
        query="Fix this code to include proper function definitions: {{code_generation.output.code}}",
        system_prompt="Generate valid Python code with proper function definitions."
    )
)

# Handle successful code execution with validation
code_execution.on(
    IsSuccess() & TextContains("result"),
    utils.send_status(
        status="analysis_complete",
        message="Data analysis completed successfully with validation"
    ),
    memory.store(memory={
        "content": "{{code_execution.output.result}}",
        "type": "code_validation_result"
    })
)

# Handle JSON parsing errors with recovery
validation_step.on(
    IsError(),
    utils.send_status(
        status="json_parse_failed",
        message="JSON parsing failed, using fallback processing"
    ),
    llm.message(
        model="claude-sonnet-4",
        query="Extract insights from this text even if it's not valid JSON: {{analysis_step.output.response}}",
        system_prompt="Extract key insights in a structured format regardless of input format."
    )
)
```

## Core Concepts

### Agents

Agents are the main containers for your AI workflows. They orchestrate multiple steps and handle complex decision-making:

```python
agent = Agent(
    name="my_automation_agent",
    description="Automates business processes with AI",
    visibility="organization"
)
```

### Steps

Steps are individual actions within an agent workflow. Each step can perform LLM calls, execute code, search the web, or interact with external services:

```python
step = agent.step(
    llm.message(
        model="claude-sonnet-4",
        query="{{user_input}}"
    ),
    depends_on=previous_step
)
```

### Result Handlers

Handle step outcomes with conditional logic using decorators or direct methods:

```python
step.on(
    IsSuccess() & GreaterThan("score", "0.8"),
    memory.store(memory={"content": "{{step.output.data}}"})
)

step.on(
    IsSuccess(),
    utils.send_status(status="completed", message="Processing finished")
)
```

### Actions

Actions are the building blocks that define what each step does:

- **LLM Actions**: Generate text, analyze content, make decisions
- **Code Execution**: Run Python scripts, process data, perform calculations
- **Memory Operations**: Store and retrieve information across workflows
- **Web Research**: Search the internet, parse websites, gather information
- **Validation Actions**: Parse JSON, validate formats, handle errors
- **Utility Actions**: Send notifications, manage state, debug workflows

## Key Features

### Intelligent Decision Making

Agents can make complex decisions based on data analysis, confidence scores, and business rules:

```python
decision_step.on(IsSuccess() & GreaterThan("risk_score", "0.7"), utils.send_status(
    status="escalated",
    message="High risk detected - human review required"
))
```

### Output Validation

Ensure AI outputs meet your standards with built-in validation tools:

```python
from erdo.conditions import TextContains, GreaterThan

# Validate content quality and format
analysis_step.on(
    IsSuccess() & TextContains("analysis") & TextContains("recommendation") & GreaterThan("confidence", "0.8"),
    memory.store(memory={"content": "{{analysis_step.output.response}}", "type": "validated"})
)

# Parse JSON output
json_step = agent.step(
    utils.parse_json(
        json="{{llm_output}}"
    )
)

# Validate specific field requirements
json_step.on(
    IsSuccess() & TextContains("status", "complete") & TextContains("data"),
    utils.send_status(status="validated", message="Output meets all requirements")
)

# Handle JSON parsing errors with recovery
json_step.on(
    IsError(),
    utils.send_status(
        status="json_error",
        message="JSON parsing failed - using fallback"
    ),
    llm.message(
        model="claude-sonnet-4",
        query="Fix this malformed JSON: {{llm_output}}",
        system_prompt="Return only valid JSON with the same content structure."
    )
)
```

### Error Handling & Recovery

Built-in error handling with retry logic and graceful degradation:

```python
process_step.on(IsError(), codeexec.execute(
    code_files=[{
        "filename": "fallback.py",
        "content": "fallback_processing({{input_data}})"
    }]
))
```

### Memory & Context Management

Persistent memory across workflow executions for context-aware processing:

```python
memory_step = agent.step(
    memory.store(memory={
        "content": "{{analysis_results}}",
        "type": "customer_insight",
        "tags": ["analysis", "{{customer_id}}"]
    })
)
```

### Scalable Deployment

Deploy agents to production with built-in monitoring, logging, and scaling:

```bash
# Deploy your agent
erdo sync

# Monitor performance
erdo logs my_agent

# Scale based on demand
erdo scale my_agent --instances 5
```
